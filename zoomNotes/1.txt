# Redis Helper Functions
def redis_get(key, default=None):
    """Get value from Redis cache"""
    if not REDIS_AVAILABLE or redis_client is None:
        return default
    try:
        data = redis_client.get(key)
        if data:
            return pickle.loads(data)
        return default
    except Exception as e:
        print(f"Redis get error for {key}: {e}")
        return default

def redis_set(key, value, expire=3600):
    """Set value in Redis cache with expiration (default 1 hour)"""
    if not REDIS_AVAILABLE or redis_client is None:
        return False
    try:
        data = pickle.dumps(value)
        redis_client.setex(key, expire, data)
        return True
    except Exception as e:
        print(f"Redis set error for {key}: {e}")
        return False

def redis_delete(key):
    """Delete key from Redis"""
    if not REDIS_AVAILABLE or redis_client is None:
        return False
    try:
        redis_client.delete(key)
        return True
    except Exception as e:
        print(f"Redis delete error for {key}: {e}")
        return False

def get_conversation_history(session_id='default'):
    """Get conversation history from Redis"""
    key = f"chat_history:{session_id}"
    return redis_get(key, [])

def save_conversation_history(session_id, history, max_messages=20):
    """Save conversation history to Redis"""
    # Keep only last max_messages
    if len(history) > max_messages:
        history = history[-max_messages:]
    key = f"chat_history:{session_id}"
    redis_set(key, history, expire=86400)  # 24 hours
    return True

# Function to calculate drawdown for combinations
def calculate_combination_drawdown(df, group_cols, min_trades=50):
    """Calculate drawdown metrics for combinations"""
    daily_pnl = df.group_by(['EntryDate'] + group_cols).agg([
        pl.sum('PnL_Adjusted').alias('DailyPnL')
    ]).sort(['EntryDate'] + group_cols)
    
    results = []
    combinations = df.group_by(group_cols).agg([
        pl.len().alias('TotalTrades')
    ]).filter(pl.col('TotalTrades') >= min_trades)
    
    for row in combinations.iter_rows(named=True):
        filters = [pl.col(col) == row[col] for col in group_cols]
        filter_expr = filters[0]
        for f in filters[1:]:
            filter_expr = filter_expr & f
        
        combo_daily = daily_pnl.filter(filter_expr).sort('EntryDate')
        
        if len(combo_daily) < 5:
            continue
        
        combo_daily = combo_daily.with_columns([
            pl.col('DailyPnL').cum_sum().alias('CumulativePnL')
        ])
        
        combo_daily = combo_daily.with_columns([
            pl.col('CumulativePnL').cum_max().alias('PeakPnL')
        ])
        
        combo_daily = combo_daily.with_columns([
            (pl.col('CumulativePnL') - pl.col('PeakPnL')).alias('Drawdown')
        ])
        
        max_peak = combo_daily['PeakPnL'].max()
        max_dd = combo_daily['Drawdown'].min()
        max_dd_pct = (max_dd / abs(max_peak) * 100) if max_peak != 0 else 0
        
        stats = combo_daily.select([
            pl.sum('DailyPnL').cast(pl.Float64).alias('TotalPnL'),
            pl.mean('DailyPnL').cast(pl.Float64).alias('AvgDailyPnL'),
            pl.std('DailyPnL').cast(pl.Float64).alias('DailyVolatility'),
            pl.len().cast(pl.Int64).alias('TradingDays')
        ])
        
        stats = stats.with_columns([
            pl.lit(max_dd, dtype=pl.Float64).alias('MaxDrawdown'),
            pl.lit(max_dd_pct, dtype=pl.Float64).alias('MaxDrawdownPct'),
            (pl.col('AvgDailyPnL') / pl.col('DailyVolatility')).cast(pl.Float64).alias('SharpeRatio')
        ])
        
        for col in group_cols:
            col_type = df[col].dtype
            stats = stats.with_columns([
                pl.lit(row[col], dtype=col_type).alias(col)
            ])
        
        results.append(stats)
    
    if results:
        return pl.concat(results)
    return None

        set2 = get_security_set(strat2, strategy_securities)
        if len(set1) == 0 and len(set2) == 0:

        union = len(set1.union(set2))
        jaccard = intersection / union if union > 0 else 0
        if jaccard > 0.3:
            security_similar_pairs.append({
                'Strategy1': int(strat1),
                'Strategy2': int(strat2),
                'JaccardSimilarity': jaccard

# AI Chat Function
def get_ai_response(user_query, conversation_history=[]):
    """Get AI response about trading data"""
    if not OPENAI_AVAILABLE:
        return "OpenAI API not available. Please install openai package and set your API key."
    
    try:
        # Get API key from environment or use placeholder
        api_key = os.getenv('OPENAI_API_KEY', '')
        if not api_key:
            return "⚠️ OpenAI API key not found. Please set OPENAI_API_KEY environment variable.\n\nFor now, here's a basic response based on your data:\n\n" + get_basic_response(user_query)
        
        # Prepare context
        data_context = prepare_data_context()
        
        # Build messages
        messages = [
            {
                "role": "system",
                "content": f"""You are a trading data analyst assistant. You help users understand their trading performance data.

Here is the current trading data context:
{data_context}

Answer questions about the trading data, provide insights, and suggest improvements based on the data shown above."""
            }
        ]
        
        # Add conversation history
        messages.extend(conversation_history)
        
        # Add current query
        messages.append({
            "role": "user",
            "content": user_query
        })
        
        # Call OpenAI API (try new API format first, fallback to old)
        try:
            # Try new API format (OpenAI v1.0+)
            client = OpenAI(api_key=api_key)
            response = client.chat.completions.create(
                model="gpt-3.5-turbo",
                messages=messages,
                max_tokens=500,
                temperature=0.7
            )
            return response.choices[0].message.content
        except (AttributeError, TypeError):
            # Fallback to old API format
            openai.api_key = api_key
            response = openai.ChatCompletion.create(
                model="gpt-3.5-turbo",
                messages=messages,
                max_tokens=500,
                temperature=0.7
            )
            return response.choices[0].message.content
        
    except Exception as e:
        return f"Error calling AI: {str(e)}\n\n" + get_basic_response(user_query)

    elif 'correlation' in query_lower or 'similar' in query_lower:

            return f"Most correlated strategy pair: Strategy {int(top_pair['Strategy1'])} and Strategy {int(top_pair['Strategy2'])} with correlation {top_pair['Correlation']:.3f}."
        return "No highly correlated pairs found in the data."
    
    elif 'win rate' in query_lower:
        return f"Overall win rate: {overall_stats['WinRate'].iloc[0]:.2f}%\nShort positions: {long_short[long_short['PositionType']=='Short']['WinRate'].iloc[0]:.1f}%\nLong positions: {long_short[long_short['PositionType']=='Long']['WinRate'].iloc[0]:.1f}%"
    
    elif 'drawdown' in query_lower:
        max_dd = daily_perf['Drawdown'].min()
        max_dd_pct = (max_dd / daily_perf['PeakPnL'].max() * 100) if daily_perf['PeakPnL'].max() > 0 else 0
        return f"Maximum drawdown: ${max_dd:,.2f} ({max_dd_pct:.2f}%)\nDate range: {daily_perf['EntryDate'].min()} to {daily_perf['EntryDate'].max()}"

# Callback for Redis Alert (checks every 10 seconds)
@app.callback(
    Output('redis-alert', 'children'),
    Input('redis-interval', 'n_intervals')
)
def check_redis_alert(n_intervals):
    """Check Redis for 'keyAlert' key every 10 seconds and display value in red"""
    if not REDIS_AVAILABLE or redis_client is None:
        return ""  # Return empty if Redis not available
    
    try:
        # Ping Redis to check connection
        redis_client.ping()
        
        # Check for 'keyAlert' key
        alert_value = redis_client.get('keyAlert')
        
        if alert_value:
            # Decode the value (assuming it's a string)
            try:
                # Try to decode as UTF-8 string
                if isinstance(alert_value, bytes):
                    alert_text = alert_value.decode('utf-8')
                else:
                    alert_text = str(alert_value)
                
                # Return the alert text in red
                return f"🚨 ALERT: {alert_text}"
            except Exception as e:
                # If decoding fails, return raw value
                return f"🚨 ALERT: {str(alert_value)}"
        else:
            # No alert found, return empty
            return ""
            
    except (redis.ConnectionError, redis.TimeoutError) as e:
        # Redis connection lost
        return f"⚠️ Redis Connection Lost: {str(e)}"
    except Exception as e:
        # Other errors
        return f"⚠️ Redis Error: {str(e)}"

# Store conversation history (in-memory fallback)
conversation_history = {}

# Callback for AI chat
@app.callback(
    [Output('chat-history', 'children'),
     Output('chat-input', 'value')],
    [Input('send-button', 'n_clicks'),
     Input('chat-input', 'n_submit')],
    [State('chat-input', 'value'),
     State('chat-history', 'children')]
)
def update_chat(n_clicks, n_submit, user_input, current_history):
    """Handle chat interactions with Redis persistence"""
    if not user_input or (not n_clicks and not n_submit):
        return current_history or [], ""
    
    # Use default session ID (could be enhanced with user sessions)
    session_id = 'default'
    
    # Get conversation history from Redis or memory
    if session_id not in conversation_history:
        conversation_history[session_id] = get_conversation_history(session_id)
    
    # Get current history or initialize
    if current_history is None:
        history_divs = []
    else:
        history_divs = current_history if isinstance(current_history, list) else [current_history]
    
    # Add user message
    user_msg = html.Div([
        html.Strong("You: "),
        html.Span(user_input)
    ], className="mb-3 text-end", style={'text-align': 'right', 'color': '#0066cc'})
    history_divs.append(user_msg)
    
    # Get AI response
    try:
        # Check Redis cache for common queries
        query_cache_key = f"ai_response:{hash(user_input.lower().strip())}"
        cached_response = redis_get(query_cache_key)
        
        if cached_response:
            ai_response = cached_response
            print(f"  ✓ Using cached AI response")
        else:
            ai_response = get_ai_response(user_input, conversation_history[session_id])
            # Cache the response for 1 hour
            redis_set(query_cache_key, ai_response, expire=3600)
        
        # Add to conversation history for context
        conversation_history[session_id].append({"role": "user", "content": user_input})
        conversation_history[session_id].append({"role": "assistant", "content": ai_response})
        
        # Keep only last 10 exchanges to avoid token limits
        if len(conversation_history[session_id]) > 20:
            conversation_history[session_id] = conversation_history[session_id][-20:]
        
        # Save to Redis
        save_conversation_history(session_id, conversation_history[session_id])
        
    except Exception as e:
        ai_response = f"Error: {str(e)}"
    
    # Add AI response
    ai_msg = html.Div([
        html.Strong("AI Assistant: "),
        html.Span(ai_response, style={'white-space': 'pre-wrap'})
    ], className="mb-3", style={'color': '#28a745'})
    history_divs.append(ai_msg)
    
    # Clear input
    return history_divs, ""

# Callback to clear Redis cache
@app.callback(
    Output('cache-status', 'children'),
    Input('clear-cache-btn', 'n_clicks'),
    prevent_initial_call=True
)
def clear_cache(n_clicks):
    """Clear all Redis cache"""
    if not REDIS_AVAILABLE or redis_client is None:
        return " (Redis not available)"
    
    try:
        # Clear specific cache keys
        cache_keys = [
            "strategy_correlation_matrix",
            "strategy_similarity_matrix",
            "top_strategies_list"
        ]
        cleared_count = 0
        for key in cache_keys:
            if redis_delete(key):
                cleared_count += 1
        
        # Clear all chat history
        try:
            keys = redis_client.keys("chat_history:*")
            if keys:
                redis_client.delete(*keys)
                cleared_count += len(keys)
        except:
            pass
        
        # Clear AI response cache
        try:
            ai_keys = redis_client.keys("ai_response:*")
            if ai_keys:
                redis_client.delete(*ai_keys)
                cleared_count += len(ai_keys)
        except:
            pass
        
        return f" → Cleared {cleared_count} keys ✓"
    except Exception as e:
        return f" → Error: {str(e)}"
